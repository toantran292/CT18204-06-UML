\paragraph{Hệ thống quản lý giải đấu Leegues}\mbox{}

\noindent
\textbf{Ưu điểm:}
\vspace{-0.5em}
\begin{itemize}[leftmargin=1.5cm, label={--}]
  \item Giao diện đẹp, dễ dùng
  \item Website hỗ trợ giao diện đa thiết bị giúp dễ dàng quản lý ở mọi nơi.
  \item Hỗ trợ sinh hình vẽ trực quan về trận đấu.
  \item Quản lý về tài chính
  \item Tích hợp gửi email tự động.

\end{itemize}

\noindent
\textbf{Nhược điểm:}
\vspace{-0.5em}
\begin{itemize}[leftmargin=1.5cm, label={--}]
  \item Không có chức năng của nhà tài trợ, thêm hình tài trợ về thi đấu
  \item Không có chức năng ứng tuyển vô đội, giải đấu.
  \item Chưa có thống kế tổng quan về giải đấu.

\end{itemize}

\noindent
\textbf{Đề xuất cải thiện:}
\vspace{-0.5em}
\begin{itemize} [leftmargin=1.5cm, label={--}]
  \item Thêm chức năng cho nhà tài trợ.
  \item Thêm thông tin của nhà tài trợ vào giải đấu.
  \item Thêm chức năng thống kế tổng quan về giải đấu.
  \item Thêm chức năng ứng tuyển vô đội thi đấu, ứng tuyển đội vào giải đấu.


\end{itemize}
