\paragraph{Hệ thống quản lý giải đấu ProLeague}\mbox{}

\textbf{Ưu điểm:}
\begin{itemize}[itemindent=1cm, label={--}]
  \item Truyền thông đến đội bóng, cầu thủ, trọng tài,... bằng SMS hoàn toàn tự động một cách nhanh chóng.
  \item Tự động tổng hợp toàn bộ số liệu chuyên môn của giải, cập nhật lên trang thông tin của giải.
  \item Toàn bộ thông tin hình ảnh của của giải đấu được quản lý trên hệ thống phần mềm, và thể hiện rõ ràng trên trang thông tin của giải đấu.
  \item Phần mềm cho phép các đội bóng tự nhập liệu thông tin đội bóng, cầu thủ, đăng ký tham gia giải một cách dễ dàng.

\end{itemize}

\textbf{Nhược điểm:}
\begin{itemize}[itemindent=1cm, label={--}]
  \item Người dùng không có tài khoản không thể xem thông tin về giải đấu.
  \item Mất nhiều thời gian để đăng ký và xác thực.
\end{itemize}

\textbf{Đề xuất cải thiện:}
\begin{itemize} [itemindent=1cm, label={--}]
  \item Hiển thị thông tin giải đấu cho người dùng không có tài khoản để người dùng có thể biết đến và theo dõi giải đấu dễ dàng hơn.
  \item Thiết kế hệ thống tạo tài khoản và tổ chức giải đấu để giải quyết các nhu cầu tạo giải đấu mới nhanh chóng hơn.
\end{itemize}
