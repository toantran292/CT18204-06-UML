\paragraph{Hệ thống quản lý giải đấu WeSport}\mbox{}

\noindent
\textbf{Ưu điểm:}
\vspace{-0.5em}
\begin{itemize}[leftmargin=1.5cm, label={--}]
  \item Tìm sân và sự kiện thể thao dễ dàng.
  \item Có chuyển đổi ngôn ngữ anh và việt.
  \item Tạo trận đấu riêng hoặc tham gia các giải đấu khác
  \item Có thể kết bạn cùng với những người cùng sở thích.
  \item Dễ dàng tạo đội và trao đổi thông tin.
  \item Hỗ trợ trên thiết bị smartphone.
  \item Có sẵn ứng dụng trên AppStore và GooglePlay.
  \item Giao diện đẹp mắt.
\end{itemize}

\noindent
\textbf{Nhược điểm:}
\vspace{-0.5em}
\begin{itemize}[leftmargin=1.5cm, label={--}]
  \item Còn thiếu nhiều môn thể thao.
  \item Thường bị lỗi khi đăng nhập và đăng ký tài khoản.
\end{itemize}

\noindent
\textbf{Đề xuất cải thiện:}
\vspace{-0.5em}
\begin{itemize} [leftmargin=1.5cm, label={--}]
  \item Sửa các lỗi khi đăng nhập và đăng ký tài khoản.
  \item Cho thêm nhiều môn thể thao.
\end{itemize}
