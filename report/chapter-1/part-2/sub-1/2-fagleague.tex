\paragraph{Hệ thống quản lý giải đấu FAGleague}\mbox{}

\textbf{Ưu điểm:}
\begin{itemize}[itemindent=1cm]
  \item Giao diện đẹp và thân thiện
  \item Sử dụng trên máy tính và Smartphone
  \item Tính năng nổi trội: dành riêng cho đơn vị tổ chức giải nên có những tính năng rất thiết thực
  \item Sử dụng những thể thức thi đấu phổ biến hiện nay: Chia cặp, đấu loại trực tiếp, thể thức league (đá vòng tròn 1 lượt hoặc 2 lượt), thể thức thi đấu Cup (chia bảng, loại trực tiếp. Đặc biệt trong vòng bảng có thể thêm bớt hoặc chỉnh sửa lịch thi đấu theo ý muốn. Ví dụ: cúp vàng, cúp bạc hoặc serie A, Serie B,  Serie C…)
  \item Giúp đội bóng quản lý các cầu thủ tham gia giải đấu.
  \item Giúp ban tổ chức giải quản lý một cách dễ dàng.

\end{itemize}

\textbf{Nhược điểm:}
\begin{itemize}[itemindent=1cm]
  \item Web xuất hiện lỗi “Resource not found.” khi bấm vào xem thông tin chi tiết các đội trong giải đấu
  \item Web còn thiếu chức năng phân loại giúp người dùng tìm kiếm các giải đấu theo thể thức thi đấu thi đấu một cách nhanh chóng
  \item Chỉ hỗ trợ cho các giải thi đấu bóng đá
  \item Danh sách tất cả các giải đấu ở trang chủ được liệt kê rất dài khiến cho người dùng khó xác định được các giải đấu đã được xem qua

\end{itemize}

\textbf{Đề xuất cải thiện:}
\begin{itemize} [itemindent=1cm]
  \item Sửa lỗi  “Resource not found.” khi bấm vào xem thông tin chi tiết các đội trong giải đấu.
  \item Thêm chức năng phân loại giải đấu theo thể thức thi đấu
  \item Hỗ trợ mở rộng cho các môn thể thao khác.
  \item Nên thêm chức năng xem trực tiếp giải đấu để đáp ứng  thêm nhu cầu được hưởng thức giải đấu
  \item Danh sách tất cả các giải đấu ở trang chủ nên được phân trang thay vì cuộn vô hạn để người dùng dễ dàng phân biệt được các giải đấu đã xem qua.

\end{itemize}
