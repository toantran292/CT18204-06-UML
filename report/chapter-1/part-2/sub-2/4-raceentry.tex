\paragraph{Hệ thống quản lý giải đấu Race Entry}\mbox{}

\textbf{Ưu điểm:}
\begin{itemize}[itemindent=1cm]
  \item Là ứng dụng miễn phí
  \item Tạo bản đồ đua miễn phí
  \item Theo dõi và xem doanh thu đánh giá theo kỳ
  \item Tích hợp Facebook
  \item Có phần đánh giá của những người tham gia cuộc đua
  \item Đề xuất những cuộc đua phù hợp với người dự thi
\end{itemize}

\textbf{Nhược điểm:}
\begin{itemize}[itemindent=1cm]
  \item Chưa hỗ trợ nhiều môn thể thao
  \item Giao diện không hoàn chỉnh
  \item Không xem được số người đã đăng ký tham gia cuộc đua
\end{itemize}

\textbf{Đề xuất cải thiện:}
\begin{itemize} [itemindent=1cm]
  \item Cần thiết kế giao diện hoàn thiện
  \item Hỗ trợ thêm các môn thể thao ở lĩnh vực khác
  \item Thêm chức năng thống kê các thí sinh tham gia cuộc đua
\end{itemize}
