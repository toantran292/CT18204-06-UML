\paragraph{Hệ thống quản lý giải đấu Copafacil}\mbox{}

\textbf{Ưu điểm:}
\begin{itemize}[itemindent=1cm]
  \item Tạo 1 giải đấu một cách nhanh chóng với các thể loại thi đấu khác nhau như:Đấu vòng, đấu cúp(chia bảng, loại trực tiếp. Có thể thêm bớt hoặc chỉnh sửa lịch thi đấu,tỉ số,cập nhật trận đấu theo ý muốn,…)
  \item Có thể theo dõi, thống kê, sắp xếp nhiều giải đấu.
  \item In xuất các file thi đấu.
  \item Cập nhật thông tin cầu thủ, đội bóng.
  \item Đăng thông tin bài báo, video về giải đấu.
\end{itemize}

\textbf{Nhược điểm:}
\begin{itemize}[itemindent=1cm]
  \item Giao diện hệ thống không thu hút người dùng
  \item Sự sắp xếp trận đấu không ngẫu nhiên.
\end{itemize}

\textbf{Đề xuất cải thiện:}
\begin{itemize} [itemindent=1cm]
  \item Thêm những chức năng phát triển về mặt quảng cáo.
  \item Cải thiện giao diện.
  \item Tự động sắp xếp giải đấu.
\end{itemize}
