\paragraph{Hệ thống quản lý giải đấu Active Network}\mbox{}

\textbf{Ưu điểm:}
\begin{itemize}[itemindent=1cm]
  \item Giao diện đẹp và thu hút người dùng
  \item Có sẵn trên web và google play store
  \item Hỗ trợ đa dạng các giải đấu thể thao
  \item Có hệ thống gửi email tích hợp sử dụng để liên lạc nhanh chóng và hiệu quả với những người tham gia giải đấu
  \item Đăng ký tài khoản một cách nhanh chóng và dễ dàng
  \item Hỗ trợ quản lí tài chính với các báo cáo tài chính chi tiết giúp ban tổ chức quản lý ngân sách hiệu quả

\end{itemize}

\textbf{Nhược điểm:}
\begin{itemize}[itemindent=1cm]
  \item Khó tiếp cận và sử dụng đối với những người mới
  \item Không có các chức năng dành cho nhà tài trợ giải đấu
  \item Xuất hiện lỗi giao diện màu nền trùng với màu chữ khiến người dùng không thể đọc rõ nội dung

\end{itemize}

\textbf{Đề xuất cải thiện:}
\begin{itemize} [itemindent=1cm]
  \item Xây dựng giao diện thân thiện với người dùng hơn
  \item Thêm chức năng dành cho các nhà tài trợ giải đấu
  \item Hoàn thiện giao diện

\end{itemize}
